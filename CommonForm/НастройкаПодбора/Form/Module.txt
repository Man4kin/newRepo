
&НаКлиенте
Процедура ОК(Команда)
	Результат = Новый Структура;
	Результат.Вставить("ЗапрашиватьКоличество", ЗапрашиватьКоличество);
	Результат.Вставить("ЗапрашиватьЦену",     ЗапрашиватьЦену);
	Закрыть(Результат);
КонецПроцедуры

&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	Если Параметры.Свойство("ЗапрашиватьКоличество") Тогда
		ЗапрашиватьКоличество = Параметры.ЗапрашиватьКоличество;
		ЗапрашиватьКоличествоСтарое= Параметры.ЗапрашиватьКоличество;
	Иначе
		Элементы.ЗапрашиватьКоличество.Доступность = Ложь;
	КонецЕсли;	
	Если Параметры.Свойство("ЗапрашиватьЦену") Тогда
		ЗапрашиватьЦену = Параметры.ЗапрашиватьЦену;
		ЗапрашиватьЦенуСтарое= Параметры.ЗапрашиватьЦену;
	Иначе
		Элементы.ЗапрашиватьЦену.Доступность = Ложь;
	КонецЕсли;	
КонецПроцедуры

&НаКлиенте
Процедура Отмена(Команда)
	Результат = Новый Структура;

	Результат.Вставить("ЗапрашиватьЦену",     ЗапрашиватьЦенуСтарое);
	Закрыть(Результат);
