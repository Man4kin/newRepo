Функция ФормаВыполненияЗадачи(Знач ЗадачаСсылка, Знач ТочкаМаршрутаСсылка) Экспорт
	Результат = Новый Структура;
	ДокСсылка = Документы.ПриходныйКассовыйОрдер.НайтиПоРеквизиту("ДокументОснование",ЗадачаСсылка.БизнесПроцесс.Основание);
	Если ДокСсылка.Пустая() Тогда
		СтрокаТаблицы = ЗадачаСсылка.Дополнительно.Найти("Аналитика","Характеристика");
		Аналитика = СтрокаТаблицы.Значение;
		СтрокаТаблицы = ЗадачаСсылка.Дополнительно.Найти("Всего","Характеристика");
		Всего = СтрокаТаблицы.Значение;
		
		СтруктураОснования = Новый Структура("Основание,Всего,Аналитика",ЗадачаСсылка.БизнесПроцесс.Основание,Всего,Аналитика);
		Результат.Вставить("ПараметрыФормы", Новый Структура("Основание,ВыполняемаяЗадача",СтруктураОснования,ЗадачаСсылка));
	Иначе
		Результат.Вставить("ПараметрыФормы", Новый Структура("КЛюч,ВыполняемаяЗадача",ДокСсылка,ЗадачаСсылка));
	КонецЕсли;	
	Результат.Вставить("ИмяФормы", "Документ.ПриходныйКассовыйОрдер.Форма.ФормаДокумента");
	Возврат Результат;
КонецФункции // ФормаВыполненияЗадачи()
