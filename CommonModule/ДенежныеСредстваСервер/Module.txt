Функция ПолучитьОстатокДенежныхСредств(ДенежнаяЯчейка,Фирма) Экспорт
	Запрос = Новый Запрос;
	Запрос.Текст = 
	"ВЫБРАТЬ ПЕРВЫЕ 1
	|	ЕСТЬNULL(ДенежныеСредстваОстатки.СуммаОстаток, 0) КАК СуммаОстаток
	|ИЗ
	|	РегистрНакопления.ДенежныеСредства.Остатки КАК ДенежныеСредстваОстатки
	|ГДЕ
	|	ДенежныеСредстваОстатки.ДенежнаяЯчейка = &ДенежнаяЯчейка
	|	И ДенежныеСредстваОстатки.Фирма = &Фирма";
	Запрос.УстановитьПараметр("ДенежнаяЯчейка",ДенежнаяЯчейка);
	Запрос.УстановитьПараметр("Фирма",Фирма);
	Выборка = Запрос.Выполнить().Выбрать();
	Выборка.Следующий();
	Возврат Выборка.СуммаОстаток;
