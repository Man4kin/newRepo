
Процедура ПередЗаписью(Отказ)
	Если Не ЗначениеЗаполнено(Автор) Тогда
		Автор = ПараметрыСеанса.ТекущийПользователь;
	КонецЕсли;
	Если Не ЗначениеЗаполнено(СозданВЦентре) Тогда
		СозданВЦентре = ПараметрыСеанса.ЭтоЦентр;
	КонецЕсли;
	Если Не ПометкаУдаления = Ссылка.ПометкаУдаления
		И Не (Автор = ПараметрыСеанса.ТекущийПользователь) Тогда
		Отказ = Истина;
	КонецЕсли;
КонецПроцедуры

Процедура ПередУдалением(Отказ)
	Отказ = Не (Автор = ПараметрыСеанса.ТекущийПользователь);
КонецПроцедуры

Процедура ПриЗаписи(Отказ)
	Если НЕ ОбменДанными.Загрузка Тогда
		Если ПараметрыСеанса.ЭтоЦентр Тогда
			Узел = ПланыОбмена.ОбменRIB.НайтиПоКоду(1);
			Если НЕ Узел.Пустая() Тогда
				ПланыОбмена.ЗарегистрироватьИзменения(Узел,ЭтотОбъект);	
			КонецЕсли;	
		КонецЕсли;	
	КонецЕсли;
КонецПроцедуры
