Процедура ПриЗаписи(Отказ)
	// здесь устанавливаем флаг шаблона по умолчанию
	Если ПоУмолчанию Тогда
		выбШаблоны = Справочники.ШаблоныЦенников.Выбрать();
		Пока выбШаблоны.Следующий() Цикл
			Если выбШаблоны.Ссылка = ЭтотОбъект.Ссылка Тогда
				Продолжить;
			КонецЕсли;
			Если выбШаблоны.ПоУмолчанию И выбШаблоны.ДляАкций = ДляАкций Тогда
				обШаблон = выбШаблоны.ПолучитьОбъект();
				обШаблон.ПоУмолчанию = Ложь;
				обШаблон.Записать();
			КонецЕсли;
		КонецЦикла; // выбШаблоны.Следующий()
	КонецЕсли; 
КонецПроцедуры

Процедура ПередЗаписью(Отказ)
	Если ЭтотОбъект.Области.Количество() = 0 И ЭтотОбъект.ОбменДанными.Загрузка = Ложь Тогда
		Сообщить("Не заданы области для печати");
		Отказ = Истина;
	КонецЕсли;
