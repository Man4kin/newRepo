		Область.ГраницаСлева = Обвести;
		Область.ГраницаСнизу = Обвести;
		Область.ГраницаСправа = Обвести;
		
		Если ОбластьТип = ВернутьЗначениеПеречисления("Текст") Или ОбластьТип = ВернутьЗначениеПеречисления("ЗначениеБД") Тогда
			Область.Текст = Текст;
			Область.Шрифт = Новый Шрифт(ШрифтИмя, Размер, Жирный, Наклонный, Ложь, Ложь);

			Область.ГоризонтальноеПоложение = ГоризонтальноеПоложение;
