
&НаКлиенте
Процедура ОбработкаКоманды(ПараметрКоманды, ПараметрыВыполненияКоманды)
	ПараметрыФормы = Новый Структура("АдресВхранилище,Режим",ПоместитьВХранилище(ПараметрКоманды[0]),"ПечатьЦенников");
	ОткрытьФорму("Обработка.ПечатьЦенниковИЭтикеток.Форма.ПечатьЦенниковИЭтикеток", ПараметрыФормы, ПараметрыВыполненияКоманды.Источник, ПараметрыВыполненияКоманды.Уникальность, ПараметрыВыполненияКоманды.Окно);
КонецПроцедуры

&НаСервере
ФУнкция ПоместитьВХранилище(Источник)
	СтруктураПараметров = Новый СТруктура;
	СтруктураПараметров.вставить("Состав",Источник.Состав.Выгрузить(,"Номенклатура,ЕдиницаИзмерения"));
	СтруктураПараметров.Вставить("ДатаДок",Источник.Дата);
	СтруктураПараметров.Вставить("Фирма",Источник.фирма);
	//СТруктураПараметров.Вставить("Валюта",Источник.валюта);
	СтруктураПараметров.Вставить("СтруктурнаяЕдиница", Источник.Склад.Владелец);
	Возврат ПоместитьВОВременноеХранилище(СтруктураПараметров);	 
КОнецФункции
