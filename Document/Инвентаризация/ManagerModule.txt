					ЕдиницаИзмеренияКодПоОКЕИ          = ЕдиницаИзмерения.Код;
				Конецесли;

				СуммаФАкт = ВыборкаСтрокТовары.СуммаФакт;				
				СуммаБух = ВыборкаСтрокТовары.СуммаУчет;
				
				ОбластьМакета.Параметры.ТоварКод =  СокрЛП(ВыборкаСтрокТовары.Номенклатура.код);
				ОбластьМакета.Параметры.ТоварНаименование = ВыборкаСтрокТовары.НоменклатураНаименование;
				ОбластьМакета.Параметры.Номер = Ном;
				ОбластьМакета.Параметры.ЕдиницаИзмеренияНаименование = ЕдиницаИзмеренияПолноеНаименование;
				ОбластьМакета.Параметры.ЕдиницаИзмеренияКодПоОКЕИ    = ЕдиницаИзмеренияКодПоОКЕИ;
				
				КоличествоФакт = ВыборкаСтрокТовары.КоличествоФакт;
				КоличествоБух = ВыборкаСтрокТовары.КоличествоУчет;
				
				Если СуммаФАкт > 0 И КоличествоФакт > 0 Тогда
					
					ОбластьМакета.Параметры.Цена = Окр(СуммаФАкт / КоличествоФакт,2);
					
				ИначеЕсли СуммаБух > 0 И КоличествоБух > 0 Тогда
					
					ОбластьМакета.Параметры.Цена = Окр(СуммаБух  / КоличествоБух,2);
					
				КонецЕсли;	
				ОбластьМакета.Параметры.ФактКоличество = КоличествоФакт;
				ОбластьМакета.Параметры.ФактСумма = СуммаФакт;
				ОбластьМакета.Параметры.БухКоличество = КоличествоБух;
				ОбластьМакета.Параметры.БухСумма = СуммаБух;
				ИтогоФактКоличествоСтр = итогоФактКоличествоСтр + КоличествоФАкт;
				ИтогоФактСуммаСтр = итогоФАктСуммаСтр + СуммаФакт;
				ИтогоБухКОличествоСтр = итогобухКоличествоСтр +КоличествоБух;
				ИтогоБухСуммаСтр = ИтогоБухСуммаСтр + СуммаБух;
				ИтогоПоОписиФактКоличествоСтр = итогоПоОписиФактКоличествоСтр + КоличествоФАкт;
				ИтогоПоОписиФактСуммаСтр = итогоПоОписиФАктСуммаСтр + СуммаФакт;
				ИтогоПоОписиБухКОличествоСтр = итогоПоОписибухКоличествоСтр +КоличествоБух;
				ИтогоПоОписиБухСуммаСтр = ИтогоПоОписиБухСуммаСтр + СуммаБух;
				ТабДок.Вывести(ОбластьМакета);
				КолвоСтрокПоСтранице = КолвоСтрокПоСтранице + 1;
			КонецЦикла;
			ОбластьИтогоПоСтранице = Макет.ПолучитьОбласть("ИтогоПоСтранице");
			ОбластьИтогоПоСтранице.Параметры.ИтогоФактКоличество = ИтогоФактКоличествоСтр; 
			ОбластьИтогоПоСтранице.Параметры.ИтогоФактСумма = ИтогоФактСуммаСтр;
			ОбластьИтогоПоСтранице.Параметры.ИтогоБухКоличество = ИтогоБухКоличествоСтр;
			ОбластьИтогоПоСтранице.Параметры.ИтогоБухСумма = ИтогоБухСуммаСтр;
			ТабДок.Вывести(ОбластьИтогоПоСтранице);
			ОбластьИтогоПоОписи.Параметры.ИтогоПоОписиФактКоличество = ИтогоПоОписиФактКоличествоСтр; 
			ОбластьИтогоПоОписи.Параметры.ИтогоПоОписиФактСумма = ИтогоПоОписиФактСуммаСтр;
			ОбластьИтогоПоОписи.Параметры.ИтогоПоОписиБухКоличество = ИтогоПоОписиБухКоличествоСтр;
			ОбластьИтогоПоОписи.Параметры.ИтогоПоОписиБухСумма = ИтогоПоОписиБухСуммаСтр;
			ТабДок.Вывести(ОбластьИтогоПоОписи);
			ОбластьИтоговПоСтранице = Макет.ПолучитьОбласть("ПодвалСтраницы");
			ОбластьИтоговПоСтранице.Параметры.КоличествоПорядковыхНомеровНаСтраницеПрописью = ОбработкаПечатисервер.КоличествоПрописью(КолвоСтрокПоСтранице);
			ОбластьИтоговПоСтранице.Параметры.ОбщееКоличествоЕдиницФактическиНаСтраницеПрописью =ОбработкаПечатисервер.КоличествоПрописью(ИтогоФАктКоличествоСтр);
			ОбластьИтоговПоСтранице.Параметры.СуммаФактическиНаСтраницеПрописью =ОбработкаПечатисервер.ФорматироватьСуммуПлатежногоДокумента(ИтогоФАктсуммаСтр)+Валюта;
			ОбластьИтоговПоСтранице = Макет.ПолучитьОбласть("ПодвалОписи");
			ОбластьИтоговПоСтранице.Параметры.ОбщееКоличествоЕдиницФактическиНаСтраницеПрописью = ОбработкаПечатисервер.КоличествоПрописью(выборкаСтрокОбщийИтог.КоличествоФакт); 
			ОбластьИтоговПоСтранице.Параметры.КоличествоПорядковыхНомеровНаСтраницеПрописью =ОбработкаПечатисервер.КоличествоПрописью(ВыборкаСтрокТовары.Количество());
			ОбластьИтоговПоСтранице.Параметры.ОбщееКоличествоЕдиницФактическиНаСтраницеПрописью = ОбработкаПечатисервер.КоличествоПрописью(ВыборкаСтрокОбщийИтог.КоличествоФакт);
			чВыборкаСтрокОбщийИтогСУммаФакт = ВыборкаСтрокОбщийИтог.СУммаФакт;
			ОбластьИтоговПоСтранице.Параметры.СуммаФактическиНаСтраницеПрописью =ОбработкаПечатисервер.ФорматироватьСуммуПлатежногоДокумента(чВыборкаСтрокОбщийИтогСУммаФакт)+валюта;
			ТабДок.Вывести(ОбластьИтоговПоСтранице);
		Конеццикла;
		
		// Выводим итоги по последней странице
		УправлениеПечатью.ЗадатьОбластьПечатиДокумента(ТабДок,НомерСтрокиНачало, ОбъектыПечати,Док.Ссылка);
		ТабДок.ВывестиГоризонтальныйРазделительСтраниц();
		НомерСтрокиНачало = ТабДок.ВысотаТаблицы+1;
		Конеццикла;
	конеццикла;
	// Зададим параметры макета
	ТабДок.ПолеСверху = 0;
	ТабДок.ПолеСлева  = 0;
	ТабДок.ПолеСнизу  = 0;
	ТабДок.ПолеСправа = 0;
